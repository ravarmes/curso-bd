CREATE VIEW para Dashboard

Simplifica consultas complexas para o Power BI:

CREATE VIEW vw_dashboard_biblioteca AS
SELECT 
  l.titulo,
  c.nome AS categoria,
  COUNT(e.id) AS total_emprestimos,
  AVG(a.nota) AS media_avaliacao
FROM livro l
LEFT JOIN categoria c ON l.categoria_id = c.id
LEFT JOIN emprestimo e ON e.livro_id = l.id
LEFT JOIN avaliacao a ON a.livro_id = l.id
GROUP BY l.id, l.titulo, c.nome;
